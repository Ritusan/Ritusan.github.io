# splitChunksPlugin

// 同步加载，默认参数
// webpack.common.js
module.export = {
    optimization: {
        splitChunks: {
            // chunks: 'async':只对异步代码生效，all是对同步和异步都生效
            chunks: 'all',
            // 引入的库或包的大小>30kb,才会做代码分割
            minSize: 30000,
            // 当一个模块至少被用(被打包之后的Chunk用)了多少次之后才进行代码分割
            minChunks: 1,
            // 同时加载的模块数最多是5个
            maxAsyncRequests: 5,
            // 入口文件做代码分割最多能分割成几个js文件
            maxInitialRequests: 3,
            // 文件名的连接符
            automaticNameDelimiter: '~',
            // cacheGroups里面配置的filename有效
            name: true,
            // 当打包同步代码时，会继续走cacheGroups(缓存组)，根据cacheGroups来决定分割出来的代码放到哪个文件里面去
            cacheGroups: {
                // 配置之后打包生成的文件名字不带vendor,如果是同步代码，不能只写vendors:false，还需要进行配置
                vendors: {
                    // 判断是否在node_modules目录下
                    test: /[\\/]node_modules[\\/]/,
                    // priority的值越大优先级越高，会打包到优先级高的组里
                    priority: -10,
                    filename: 'vendors.js'
                },
                default: {
                    priority: -20,
                    // 如果一个模块已经被打包过，那么再次遇到这个模块的时候就忽略打包
                    reuseExistingChunk: true,
                    filename: 'common.js'
                },
            }
        }
    }
}
