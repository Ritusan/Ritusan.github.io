    // loader来处理对应后缀结尾的文件

import React, { Component } from 'react'
import { BrowserRouter,Route } from 'react-router-dom'
import ReactDom from 'react-dom'

class App extends Component {
    render(){
        return (
            <BrowserRouter>
                <div>
                    <Route path='/' exact component={Home} />

        )
    }
}

// .eslintrc.js
module.export = {
    "extends": "airbnb",
    // 需要安装babel-eslint
    "parser": "babel-eslint",
    "rules": {
        // 排除react/prefer-stateless-function
        "react/prefer-stateless-function": 0,
        "react/jsx-filename-extension": 0,
    },
    "globals": {
        document: false
    }
}

module: {
    rules: [
        {
            test: /\.js$/,
            exclude: /node_modules/,
            // loader的执行顺序是从后到前
            use: [
                {
                    loader: 'babel-loader'
                },
                {
                    loader: 'eslint-loader',
                    options: {
                        // 如果代码中有一些浅显的问题，会自动帮你做修复，降低eslint对打包过程性能的损耗

                    }
                }
            ]
        }
    ]
},
devServer: {
    // 在浏览器弹出层提示错误
    overlay: true
}

# Webpack性能优化

# 提升Webpack打包速度

  1. 跟上技术的迭代(升级Node,Npm,Yarn版本)

  2. 在尽可能少的模块上应用Loader

    module.export = {
    module: {
        rules: [
            {
                 test: /\.js$/,
                 // exclude: /node_modules/,
                 // 去约定只有在某一个文件夹下才使用某一个loader，从而降低loader被频繁执行的次数
                 include: path.resolve(__dirname,'../src'),
                 use: [
                     {
                         loader: 'babel-loader'
                     }
                 ]
             }
        ]
    }
}

  3. Plugin尽可能精简并确保可靠
    在开发环境不需要OptimizeCSSAssetsPlugin压缩代码，从而减少打包过程中压缩代码的时间。

  4. resolve参数合理配置

    module.export = {
    resolve: {
        // 当引入其它目录下的模块时，如果不写文件的后缀，会去对应的目录下先找以js和jsx结尾的文件
        // css和图片文件不建议配置在extensions中
        extensions: ['.js','.jsx'],
        // 当引入一个目录下的未知文件时，首先会去找名字为index的文件，其次是child(可以自己设置)
        mainFiles: ['index','child'],
        // alias别名,当看到abc路径的时候，实际上是child目录的别名
        alias: {
            abc: path.resolve(__dirname,'../src/child')
        }
    },
    module: {
        rules: [
             {
                 // 以js或jsx结尾的都会走babel-loader
                 test: /\.jsx?$/,
                 include: path.resolve(__dirname,'../src'),
                 use: [
                     {
                         loader: 'babel-loader'
                     }
                 ]
             }
        ]
    }
}

    // webpack.common.js
const path = require('path')
const fs = require('fs')
const webpack = require('webpack')
const HtmlWebpackPlugin = require('html-webpack-plugin')
const CleanWebpackPlugin = require('clean-webpack-plugin')
const AddAssetHtmlWebpackPlugin = require('add-asset-html-webpack-plugin')

const plugins = [
    new HtmlWebpackPlugin({
        template: 'src/index.html'
    }),
    new CleanWebpackPlugin(['dist'],{
        root: path.resolve(__dirname,'../')
    })
]

const files = fs.readdirync(path.resolve(__dirname,'../dll'))
// console.log(files)
files.forEach(file=>{
    if(/.*\.dll.js/.test(file)){
        plugins.push(new AddAssetHtmlWebpackPlugin({
           filepath: path.resolve(__dirname,'../dll',file)
       }))
    }
    if(/.*\.manifest.json/.test(file)){
        plugins.push(new webpack.DllReferencePlugin({
           manifest: path.resolve(__dirname,'../dll',file)
       }))
    }
})

module.export = {
   plugins: plugins
}

# 多页面打包配置

// webpack.common.js
module.export = {
    entry: {
        main: './src/index.js',
        list: './src/list.js'
    },
    plugins: [
        new HtmlWebpackPlugin({
            template: 'src/index.html',
            filename: 'index.html',
            // 要引入的chunk
            chunks: ['runtime','vendors','main']
        }),
        new HtmlWebpackPlugin({
            template: 'src/index.html',
            filename: 'list.html',
            chunks: ['runtime','vendors','list']
        }),
    ]
}
